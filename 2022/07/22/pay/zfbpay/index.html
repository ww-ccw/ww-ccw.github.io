集成支付宝支付流程

支付宝支付代码流程

一、注册支付宝开放平台

可以使用正式环境也可以使用沙箱环境。支付宝相关流程 相关文档详见

二、开发环境搭建

  1. 引入maven开发包sdk

    <dependency>
  <groupId>com.alipay.sdk</groupId>
  <artifactId>alipay-sdk-java</artifactId>
  <version>4.34.0.ALL</version>
</dependency>

  2. 配置配置文件

    # 支付宝支付相关参数

# 应用ID,您的APPID，收款账号既是您的APPID对应支付宝账号
alipay.app-id=

# 商户PID,卖家支付宝账号ID
alipay.seller-id=

# 支付宝网关 沙箱环境和正式环境不同
alipay.gateway-url=

# 商户私钥，您的PKCS8格式RSA2私钥
alipay.merchant-private-key=
# 支付宝公钥,查看地址：https://openhome.alipay.com/platform/keyManage.htm 对应APPID下的支付宝公钥
alipay.alipay-public-key=
# 接口内容加密秘钥，对称秘钥
alipay.content-key=

# 支付成功后跳转到的页面，同步通知页面路径
#例：http://localhost:8080/#/success
alipay.return-url=

# 服务器异步通知页面路径  需http://格式的完整路径，不能加?id=123这类自定义参数，必须外网可以正常访问（内网穿透）
# 注意：每次重新启动ngrok，都需要根据实际情况修改这个配置
# 例：https://2530-183-217-31-109.ap.ngrok.io/api/ali-pay/trade/notify
alipay.notify-url=

  3. 注入AlipayClient的bean

    @Configuration
//加载配置文件
@PropertySource("classpath:alipay-sandbox.properties")
public class AlipayClientConfig {
    
    @Resource
    private Environment config;
    /**
     * 支付宝接口配置文件
     * @return
     * @throws AlipayApiException
     */
    @Bean
    public AlipayClient alipayClient() throws AlipayApiException {
        AlipayConfig alipayConfig = new AlipayConfig();

        //设置网关地址
        alipayConfig.setServerUrl(config.getProperty("alipay.gateway-url"));
        //设置应用Id
        alipayConfig.setAppId(config.getProperty("alipay.app-id"));
        //设置应用私钥
        alipayConfig.setPrivateKey(config.getProperty("alipay.merchant-private-key"));
        //设置请求格式，固定值json
        alipayConfig.setFormat(AlipayConstants.FORMAT_JSON);
        //设置字符集
        alipayConfig.setCharset(AlipayConstants.CHARSET_UTF8);
        //设置支付宝公钥
        alipayConfig.setAlipayPublicKey(config.getProperty("alipay.alipay-public-key"));
        //设置签名类型
        alipayConfig.setSignType(AlipayConstants.SIGN_TYPE_RSA2);
        //构造client
        AlipayClient alipayClient = new DefaultAlipayClient(alipayConfig);

        return alipayClient;
    }
}

三、代码实现

1、统一收单下单并支付页面接口

controller

@ApiOperation("统一收单下单并支付页面接口的调用")
@PostMapping("/trade/page/pay/{productId}")
public R tradePagePay(@PathVariable long productId) {
    log.info("统一收单下单并支付页面接口的调用");
    
    //支付宝开发者平台接受 request 请求后，会为开发者生成一个html形式的form表单，包含自动提交的脚本
    String formStr = aliPayService.tradeCreate(productId);
    //将form表单字符串返回给前端程序，之后前端会调用自动提交脚本，进行表单提交
    //此时，表单会自动提交到action属性所指向的支付宝开发者平台中，从为用户展现一个支付页面
    return R.ok().data("formStr", formStr);
}

service

@Transactional(rollbackFor = Exception.class)
@Override
public String tradeCreate(Long productId) {
    try {
        //生成订单
        log.info("生成订单");
        //创建订单
        OrderInfo orderInfo = orderInfoService.createOrderByProductId(productId);
        
        //获得支付宝支付接口
        AlipayTradePagePayRequest request = new AlipayTradePagePayRequest();
        
        //配置需要的公共参数
        
        //支付成功后需要回调的地址
        request.setNotifyUrl(config.getProperty("alipay.notify-url"));
        //支付成功后返回的页面
        request.setReturnUrl(config.getProperty("alipay.return-url"));
        
        //组装当前业务方法的请求参数
        JSONObject bizContent = new JSONObject();
        //订单号
        bizContent.put("out_trade_no", orderInfo.getOrderNo());
        //支付金额
        BigDecimal total = new BigDecimal(orderInfo.getTotalFee().toString()).divide(new BigDecimal("100"));
        bizContent.put("total_amount", total);
        //订单标题
        bizContent.put("subject", orderInfo.getTitle());
        //与支付宝签约的产品码名称
        bizContent.put("product_code", "FAST_INSTANT_TRADE_PAY");
        
        //将参数装入支付宝接口
        request.setBizContent(bizContent.toString());
        //执行请求，调用支付宝接口
        AlipayTradePagePayResponse response = alipayClient.pageExecute(request);
        
        if (response.isSuccess()) {
            log.info("调用成功,返回结果");
        } else {
            log.info("调用失败,返回结果");
        }
        return response.getBody();
    } catch (AlipayApiException e) {
        e.printStackTrace();
        throw new RuntimeException("创建支付交易失败");
    }
    
}

2、异步通知代码编写

​ 必须要有外网地址—>内网穿透

​ controller：必须经过如下验证步骤，且返回最终failure或success，否则在正式环境下，支付宝会持续回调

@ApiOperation("支付通知")
   @PostMapping("/trade/notify")
   public String tradeNotify(@RequestParam Map<String, String> paramsMap) {
       log.info("支付通知正在执行");
       log.info("通知参数 ====> {}", paramsMap);
       String result = "failure";
       try {
         	 // TODO 验签成功后，按照支付结果异步通知中的描述，对支付结果中的业务内容进行二次校验，校验成功后在response中返回success并继续商户自身业务处理，校验失败返回failure 
           //调用SDK验证签名
           boolean signVerified = AlipaySignature.rsaCheckV1(paramsMap, config.getProperty("alipay.alipay-public-key"),
                   AlipayConstants.CHARSET_UTF8, AlipayConstants.SIGN_TYPE_RSA2);
           
           if (!signVerified) {
               log.error("支付成功，验签失败");
               return result;
           }
           
           //1、获得订单号
           String outTradeNo = paramsMap.get("out_trade_no");
           OrderInfo order = orderInfoService.getOrderByOrderNo(outTradeNo);
           if (order == null) {
               log.error("订单不存在");
               return result;
           }
           //2、比较金额
           String totalAmount = paramsMap.get("total_amount");
           int totalAmountInt = new BigDecimal(totalAmount).multiply(new BigDecimal(100)).intValue();
           int totalFee = order.getTotalFee();
           if (totalFee != totalAmountInt) {
               log.error("金额校验失败");
               return result;
           }
           //3、校验通知中的 seller_id（或者 seller_email) 是否为 out_trade_no 这笔单据的对应的操作方
           String sellerId = paramsMap.get("seller_id");
           String sellerIdProperty = config.getProperty("alipay.seller-id");
           if (!sellerId.equals(sellerIdProperty)) {
               log.error("商家pid校验失败");
               return result;
           }
           //4、校验app_id
           String appIdProperty = config.getProperty("alipay.app-id");
           String appId = paramsMap.get("auth_app_id");
           if (appId != appIdProperty) {
               log.error("app_id校验失败");
               return result;
           }
           
           String tradeStatus = paramsMap.get("trade_status");
           if ("TRADE_SUCCESS".equals(tradeStatus)) {
               log.error("支付未成功");
               return result;
           }
           
           //处理业务 修改订单状态 记录支付日志
           aliPayService.processOrder(paramsMap);
           
           result = "success";
           log.info("支付成功，验签成功");
       } catch (AlipayApiException e) {
           e.printStackTrace();
       }
        // TODO 验签失败则记录异常日志，并在response中返回failure.
       return result;
   }

3、取消订单

注意：针对⼆维码⽀付，只有经过扫码的订单才在⽀付宝端有交易记录。针对⽀付宝账号⽀付，只有经 过登录的订单才在⽀付宝端有交易记录。

//传入订单号
private void closeOrder(String orderNo) {

    try {
        log.info("关单接口的调用，订单号 ===> {}", orderNo);
        //支付宝取消订单接口
        AlipayTradeCloseRequest request = new AlipayTradeCloseRequest();
        //设置参数,订单号
        JSONObject bizContent = new JSONObject();
        bizContent.put("out_trade_no", orderNo);
        //装入参数
        request.setBizContent(bizContent.toString());
        //发出请求
        AlipayTradeCloseResponse response = alipayClient.execute(request);

        if(response.isSuccess()){
            log.info("调用成功，返回结果 ===> " + response.getBody());
        } else {
            log.info("调用失败，返回码 ===> " + response.getCode() + ", 返回描述 ===> " + response.getMsg());
            //throw new RuntimeException("关单接口的调用失败");
        }

    } catch (AlipayApiException e) {
        e.printStackTrace();
        throw new RuntimeException("关单接口的调用失败");
    }
}

4、主动查询订单支付情况

如果支付宝未能及时返回支付结果，需要主动向支付宝查询，可以单独创建一个线程每个一段时间从数据库中取出未支付的订单判断是否已经支付，如果未支付就关闭订单

/**
* 查询订单
* @param orderNo 订单号
* @return 返回订单查询结果，如果返回null则表示支付宝端尚未创建订单
*/
@Override
public String queryOrder(String orderNo) {

    try {
        log.info("查单接口调用 ===> {}", orderNo);
		//获得查询接口
        AlipayTradeQueryRequest request = new AlipayTradeQueryRequest();
        //设置参数
        JSONObject bizContent = new JSONObject();
        bizContent.put("out_trade_no", orderNo);
        //装入参数
        request.setBizContent(bizContent.toString());
		//发起请求
        AlipayTradeQueryResponse response = alipayClient.execute(request);
        if(response.isSuccess()){
            log.info("调用成功，返回结果 ===> " + response.getBody());
            return response.getBody();
        } else {
            log.info("调用失败，返回码 ===> " + response.getCode() + ", 返回描述 ===> " + response.getMsg());
            //throw new RuntimeException("查单接口的调用失败");
            return null;//订单不存在
        }
    } catch (AlipayApiException e) {
        e.printStackTrace();
        throw new RuntimeException("查单接口的调用失败");
    }
}
