AQS

AQS（AbstractQueuedSynchronizer）

基本组成

  • Node节点（内部类）存放Thread
    • waitState：队列中其他线程的状态-1为阻塞
    • 头节点
    • 尾节点
    • 锁类型（独占还是共享）
    • 前指针
    • 尾指针
  • CLH（双向队列）由Node组成，尾部入队，头部出队
  • state（锁状态）0表示无人占有

//源码
public class ReentrantLock implements Lock, java.io.Serializable {

    public void lock() {
        sync.lock();	//sync内部类继承于AQS
    }
    
    abstract static class Sync extends AbstractQueuedSynchronizer {
        abstract void lock(); //由子类实现，子类分别为FairSync（公平锁）、NonfairSync（非公平锁）
    }
    
    //以非公平锁为例
    static final class NonfairSync extends Sync {
	
        final void lock() {
            if (compareAndSetState(0, 1))	//通过CAS比较并修改state
                setExclusiveOwnerThread(Thread.currentThread());//成功则设置当前线程为线程所有者
            else
                acquire(1);					//如果未成功调用
        }
    }
    //见下注
    public final void acquire(int arg) {
        if (!tryAcquire(arg) &&				//注1：尝试抢占锁，如果失败返回的false取反为true	
            //注2：先将当前节点设置独占并加入等待队列，注3：将当前队列阻塞，线程将阻塞在这里
            acquireQueued(addWaiter(Node.EXCLUSIVE), arg))	
            selfInterrupt();
    }
}

public abstract class AbstractQueuedSynchronizer extends AbstractOwnableSynchronizer implements java.io.Serializable {  
	public final void acquire(int arg) {
        if (!tryAcquire(arg) &&
            acquireQueued(addWaiter(Node.EXCLUSIVE), arg))
            selfInterrupt();
    }
    
    protected boolean tryAcquire(int arg) {
        throw new UnsupportedOperationException();		//直接抛异常是为了强制子类必须实现
    }

public class ReentrantLock implements Lock, java.io.Serializable {

     static final class NonfairSync extends Sync {
        protected final boolean tryAcquire(int acquires) {	//NonfairSync的实现
            return nonfairTryAcquire(acquires);				//回到父类Sync的实现，执行非公平的抢占
        }
    }
    
    abstract static class Sync extends AbstractQueuedSynchronizer {
        final boolean nonfairTryAcquire(int acquires) {
            final Thread current = Thread.currentThread();	//获得当前线程
            int c = getState();							//得到锁的状态，0表示无人抢占
            if (c == 0) {
                if (compareAndSetState(0, acquires)) {		//cas比较并交换state
                    setExclusiveOwnerThread(current);		//如果成功将当前线程设置为抢占线程
                    return true;						  //返回true
                }
            }
            else if (current == getExclusiveOwnerThread()) {	//判断当前线程是否为抢占线程--可重入锁
                int nextc = c + acquires;						//如果是，则将state加1
                if (nextc < 0) // overflow						
                    throw new Error("Maximum lock count exceeded");
                setState(nextc);								//将state设置为新值
                return true;
            }
            return false;										//如果上述都未成功返回false
        }
    }   
}

public abstract class AbstractQueuedSynchronizer extends AbstractOwnableSynchronizer implements java.io.Serializable { 
	private Node addWaiter(Node mode) {
        //将当前节点封装为一个新节点，按照传入的参数设置为独占（或共享）
        Node node = new Node(Thread.currentThread(), mode);
        //获得同步器指向的尾节点
        Node pred = tail;
        if (pred != null) {
            //如果尾节点不为空，直接加入---->见图2
            node.prev = pred;
            if (compareAndSetTail(pred, node)) {
                pred.next = node;
                return node;
            }
        }
        //如果尾节点为空，进入初始化--->见图1
        enq(node);
        return node;
    }
    
    /**
    * 尾节点为空时，新建一个节点作为头节点，其中不加任何东西，只作为一个头（哨兵节点）
    * 然后将下个节点设置为当前线程的节点
    */
    private Node enq(final Node node) {
        //自旋设置
        for (;;) {
            Node t = tail;
            if (t == null) { // Must initialize
                if (compareAndSetHead(new Node()))
                    tail = head;
            } else {
                node.prev = t;
                if (compareAndSetTail(t, node)) {
                    t.next = node;
                    return t;
                }
            }
        }
    }
}

注3

public abstract class AbstractQueuedSynchronizer extends AbstractOwnableSynchronizer implements java.io.Serializable { 
    
    final boolean acquireQueued(final Node node, int arg) {
        //取消	
        boolean failed = true;
        try {
            //终止
            boolean interrupted = false;
            //自旋
            for (;;) {
                //获得前节点
                final Node p = node.predecessor();
                //如果前节点是头节点且尝试抢占成功，
                if (p == head && tryAcquire(arg)) {
                    setHead(node);
                    p.next = null; // help GC
                    failed = false;
                    return interrupted;
                }
                //未抢占成功
                if (shouldParkAfterFailedAcquire(p, node) &&//先false，再true，cas的方式设置waitStatus=-1
                    //未抢占到锁的线程将阻塞在这里
                    parkAndCheckInterrupt())					//使用LockSupport将当前线程设置为阻塞返回true
                    interrupted = true;							
            }
        } finally {
            if (failed)
                cancelAcquire(node);
        }
    }
   //---------------------------------shouldParkAfterFailedAcquire(p, node)---------------------------------------
    //第一次返回true，将当前线程的waitStatu设置为-1，如果再次访问返回true
    private static boolean shouldParkAfterFailedAcquire(Node pred, Node node) {
        //获得当前线程状态，初始是0
        int ws = pred.waitStatus;
        //等于-1？
        if (ws == Node.SIGNAL)
            return true;
        //大于0？
        if (ws > 0) {
            do {
                node.prev = pred = pred.prev;
            } while (pred.waitStatus > 0);
            pred.next = node;
        } else {
          	//使用cas将
            compareAndSetWaitStatus(pred, ws, Node.SIGNAL);
        }
        return false;
    }
    
    //--------------------------parkAndCheckInterrupt()-----------------------------------------------------------
    private final boolean parkAndCheckInterrupt() {
        LockSupport.park(this);				//将当前线程设置为阻塞，程序不再往下走
        return Thread.interrupted();
    }
    
}

public abstract class AbstractQueuedSynchronizer extends AbstractOwnableSynchronizer implements java.io.Serializable { 
    
	public final boolean release(int arg) {
        if (tryRelease(arg)) {			//注1：尝试释放锁，同时设置state、占有线程
            //释放成功，获得头节点
            Node h = head;
            if (h != null && h.waitStatus != 0)
                unparkSuccessor(h);		//唤醒头节点的下一个节点
            return true;
        }
        return false;
    }
    
}

abstract static class Sync extends AbstractQueuedSynchronizer {
	//尝试取消，每次只会将state减1，所以需要加锁几次，释放几次
    protected final boolean tryRelease(int releases) {
        //获取新的state
        int c = getState() - releases;
        if (Thread.currentThread() != getExclusiveOwnerThread())
            //如果当前线程不是持有线程则抛出异常
            throw new IllegalMonitorStateException();
        boolean free = false;
        if (c == 0) {
            //如果新state为0
            free = true;
            //将持有线程设置为null
            setExclusiveOwnerThread(null);
        }
        //设置新的state，每次只会将state减1，所以需要加锁几次，释放几次
        setState(c);
        return free;
    }
}

public abstract class AbstractQueuedSynchronizer extends AbstractOwnableSynchronizer implements java.io.Serializable { 
    
	private void unparkSuccessor(Node node) {
		//先设置头节点的waitStatus = 0
        int ws = node.waitStatus;
        if (ws < 0)
            compareAndSetWaitStatus(node, ws, 0);
		//获取头节点的下一个节点
        Node s = node.next;
        if (s == null || s.waitStatus > 0) {
            s = null;
            for (Node t = tail; t != null && t != node; t = t.prev)
                if (t.waitStatus <= 0)
                    s = t;
        }
        //如果下一个节点不为null，则唤醒它
        if (s != null)
            LockSupport.unpark(s.thread);
    }
}

最后回到线程阻塞处

public abstract class AbstractQueuedSynchronizer extends AbstractOwnableSynchronizer implements java.io.Serializable { 
    
    private final boolean parkAndCheckInterrupt() {
        LockSupport.park(this);				//将当前线程设置为阻塞，程序不再往下走
        return Thread.interrupted();		//1、解除阻塞，从这里开始，当前线程没有被中断，返回false
    }
    
    final boolean acquireQueued(final Node node, int arg) {
        boolean failed = true;
        try {
            boolean interrupted = false;
            for (;;) {							//3、自旋
                final Node p = node.predecessor();	//获取当前节点的前节点
                if (p == head && tryAcquire(arg)) {	//4、尝试获取锁，获取成功
                    setHead(node);				   //5、将当前节点设置为头节点--->见图1
                    p.next = null; // help GC
                    failed = false;
                    return interrupted;
                }
                if (shouldParkAfterFailedAcquire(p, node) &&
                    parkAndCheckInterrupt())		//2、阻塞在这里，解除阻塞返回false
                    interrupted = true;				
            }
        } finally {
            if (failed)
                cancelAcquire(node);				
        }
    }
    
}
